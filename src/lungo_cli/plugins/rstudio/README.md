A Lungo plugin.

```yaml
# ========

# ========

#   # Settings for the RStudio plugin
#   rstudio:

#     # Enable or disable the plugin
#     # Type: boolean
#     # Default: true
#     # Required: no
#     enabled: true

#     # Password for users inside the container
#     # Type: string
#     # Default: (randomly generated)
#     # Required: no
#     password: passwd
```
